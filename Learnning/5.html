  • class Unit:
    def __init__(self, name, hp):
    self.name = name
    self.hp = hp
  • class AttackUnit(Unit):
    def __init__(self, name, hp, damage):
    Unit.__init__(self, name, hp)
    self.damage = damage

    # 2개이상의 부모 클래스로 부터 상속받는 다중 상속도 존재
    # 부보클래스로부터 상속받은 메소드를 자식클래스에서 새롭게 재정의하는것을 오버로딩이라고 한다
    # __init__ 함수내에서 pass라고 써두면 에러없이 아무것도 안하고 그냥 넘어가진다.
    # Unit.__init__(self, name, hp)부분은 super().__init__(name, hp)로 바꿔쓸수 있다.
    # 다중상속을 받을때는 super를 쓸수 없다.
